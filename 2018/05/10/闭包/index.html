闭包

闭包就是指在函数中返回一个函数并赋值给一个外部变量，
这样函数内部的值就会被外部变量引用从而不会被垃圾回收清除，
从而达到保存并访问函数内部的变量的效果
function add(){
var a = 1;
return function(){
return ++a;
}
}
var a1 = new add();
a1();
a1();
var a2 = new add();
a2();

柯理化
function add(x){
return function(y){
return function(z){
return x+y+z;
}
}
}
var a12 = add(1)(2);
a12(3);
a12(5);

创建私有成员
function numberadd(){
var a = 0;
function add(){
return ++a;
}
function reduce(){
return –a;
}
return {
ad:add,
re:reduce
}
}
var a1 = new numberadd();
a1.ad();
a1.re();

可以避免全局变量污染，模块化代码

闭包在处理速度和内存消耗方面对脚本有影响
